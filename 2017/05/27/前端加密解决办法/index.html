import CryptoJs from ‘crypto-js’
前端和ios的padding是Pkcs7，java的padding是Pkcs5
let key = CryptoJs.enc.Utf8.parse(“:mU@-\”S61Eujx$c”);
let iv = CryptoJs.enc.Utf8.parse(‘:mU@-\”S61Eujx$c’);
let aes = {
加密
aesEncrypt(plaintext) {
return CryptoJs.AES.encrypt(plaintext, key, {mode: CryptoJs.mode[‘ECB’], padding: CryptoJs.pad[‘Pkcs7’], iv: iv}).toString();
}
解密
aesDecrypt(ciphertext) {
let plainBytes = CryptoJs.AES.decrypt(ciphertext, key, {mode: CryptoJs.mode[‘ECB’], padding: CryptoJs.pad[‘Pkcs7’], iv: iv});

}
}

key值编码

1、需要下载的代码库
npm install crypto-js –save-dev
npm install utf8-bytes

2、代码编写
crypto.js
let CryptoJs = require(‘crypto-js’),
bytes = require(‘utf8-bytes’);

class crypto {
constructor(key, iv = key, mode = ‘ECB’, padding = ‘ZeroPadding’) {
this.key = key; //传入的key值
this.iv = iv;
this.mode = mode;
this.padding = padding;
this.skey = CryptoJs.enc.Utf8.parse(key);
this.siv = CryptoJs.enc.Utf8.parse(iv);
}

encrypt(plaintext) {
let length = (16 - bytes(plaintext).length % 16) % 16;
for (let i = 0; i < length; i++) {
plaintext += ‘ ‘;c
}
plaintext = CryptoJs.enc.Utf8.parse(plaintext);

return CryptoJs.AES.encrypt(plaintext, this.skey, {
  mode: CryptoJs.mode[this.mode],
  padding: CryptoJs.pad[this.padding],
  iv: this.siv
}).toString();

}

decrypt(ciphertext) {
let plainBytes = CryptoJs.AES.decrypt(ciphertext, this.skey, {
mode: CryptoJs.mode[this.mode],
padding: CryptoJs.pad[this.padding],
iv: this.siv
});

return CryptoJs.enc.Utf8.stringify(plainBytes).toString();

}
}

module.exports = crypto;

整体编码代码

1、需下载的代码库
npm install md5 –save-dev
npm install sha256 –save-dev
npm install base64 –save-dev

2、代码编写
dataParser.js
let md5 = require(‘md5’),
sha256 = require(‘sha256’),
gateway_salt = ‘/H0w<c|:dYV(.q)b’,
base64 = require(‘base-64’),
CryptoJs = require(‘./crypto’);

class dataParser {
//new dataParser需要传入linkid,token
constructor(linkid = ‘’, token = ‘’) {
this.linkid = linkid;
this.token = token;
}
//编译key
get key() {

let key = sha256(this.token + 'eroad2016');
let index = Number(this.linkid.match(/[0-9]/g).pop());

key = key.substring(index, index + 16);

return key;

}

get crypt() {
return new CryptoJs(this.key);
}
//编码
encryptData(data = {}) {

let crypt = this.crypt;

let encryptedData = crypt.encrypt(JSON.stringify(data)),
  MD5Data = md5(encryptedData),
  MD5DataWithSalt = md5(MD5Data + this.key + gateway_salt);
//返回的值都是直邮data和summary
return {
  data: encryptedData,
  summary: MD5DataWithSalt
}

}
//解码
decryptData(data) {

if (!(data.data && data.summary)) {
  return data;
}

let key = this.key,
  encryptData = data.data,
  MD5Data = md5(encryptData),
  MD5DataWithSalt = md5(MD5Data + key + gateway_salt);

if (data.summary === MD5DataWithSalt) {

  return this.crypt.decrypt(encryptData);
} else {

  throw new Error('签名不一致，解密失败');
}

}
//需要传入到headers里面
getIdentification(username, obj = {
“type”: “basic”,
“username”: username,
“imei”: “869336029872296”,
“language”: “zh”,
“app_channel”: “ehr”,
“info”: {
‘systemVersion’: 9.1,
‘systemModel’: ‘PC’,
“deviceModel”: “M5”,
“manufacturer”: “GiONEE”
},
“company_code”: “HULB”,
version: ‘2.0.0’
}) {
obj.linkid = this.linkid;

return base64.encode(JSON.stringify(obj));

}
}

module.exports = dataParser;

调用实例 (vue方式)

//getToken
import axios from ‘axios’
import dataParser from ‘./../../untils/dataParser’
import linkid from ‘./../../untils/linkid’
import aes from ‘./../../untils/aes’
export default {
data () {
return {
error: false,
token: ‘’
}
},
methods: {
getToken() {
let parser = new dataParser(linkid)
let that = this
axios({
headers: {
identification: parser.getIdentification()
},
url: ‘/get_token’,
method: ‘POST’,
responseType: ‘json’
}).then(function (res) {
that.token = res.data.data.token
})
},
login () {
let that = this
let username = document.getElementById(‘username’).value
let password = document.getElementById(‘password’).value
password = aes.aesEncrypt(password)
let parsers = new dataParser(linkid, that.token)
let data = {
“username”: username,
“password”: password
}
let params = parsers.encryptData(data)
axios({
headers:{
identification: parsers.getIdentification(username)
},
data: params,
method: ‘post’,
url: ‘/mjson/login_new’
}).then(function (res) {
let data = JSON.parse(parsers.decryptData(res.data))
if (data.code === 0) {
that.error = false
window.location.hash = ‘/home’
} else if (data.code === 100) {
document.getElementById(‘password’).value = ‘’
that.error = true
}
})
}
},
mounted () {
document.title = ‘登陆’
this.getToken()
}
}
